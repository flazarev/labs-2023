#include <stdio.h>
#include <stdbool.h>
#define UINT_MAX 0xFFFFFFFFF

int IsSeparator(int c) {
    return (c == ' ') || (c == ',') || (c == '\n') || (c == '\t') || (c == EOF);
}

int main() {
    bool flag = false;
    int digit = EOF;
    int num = 0;
    while (1) {
        digit = getchar();
        if (!IsSeparator(digit)) {
            num = (num * 10 + (digit - '0'));
            flag = true;
        } else if (digit == EOF) {
            break;
        } else if (flag == true) {
            flag = false;
            int reverse_num = UINT_MAX - num;
            printf("%lld\n", reverse_num);
        }
        }
    return 0;
}
